go1.13.10

1) unzip

rename:

cd /usr/local/go

2) set path

vi ~/.bashrc

export GOROOT=/usr/local/go/1.13

export GOPATH=/home/tdtc/go

PATH=$PATH:$GOROOT/bin:$GOPATH/bin

source ~/.bashrc

test
